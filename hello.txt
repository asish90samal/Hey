W   = 56          # column width
SEP = "│"         # vertical divider character

def _p(s: str, w: int = W) -> str:
    """Pad / truncate string to exactly w chars."""
    return s[:w].ljust(w)

def _lines(text: str, w: int = W) -> list[str]:
    return textwrap.wrap(text, w) or [""]

def _row(l: str, r: str) -> None:
    ll = _lines(l);  rl = _lines(r)
    for i in range(max(len(ll), len(rl))):
        lx = ll[i] if i < len(ll) else ""
        rx = rl[i] if i < len(rl) else ""
        print(f" {_p(lx)} {SEP} {_p(rx)}")

def _hdr(title: str) -> None:
    bar = "═" * (W * 2 + 4)
    print(f"\n{bar}\n  {title}\n{bar}")

def _div(c: str = "─") -> None:
    print(f" {c*W} {SEP} {c*W}")

def _banner(l: str, r: str) -> None:
    print(f" {_p(l)} {SEP} {_p(r)}")

# ANSI colours
BOLD  = "\033[1m";  GRAY  = "\033[90m"; RESET = "\033[0m"
RED   = "\033[91m"; YEL   = "\033[93m"; GRN   = "\033[92m"

def _col(s: str, c: str) -> str:
    return f"{c}{s}{RESET}"

def _bar(v: float, width: int = 20) -> str:
    filled = int(round(v * width))
    return "█" * filled + "░" * (width - filled)

def _decision_color(d: str) -> str:
    return {
        "ALERT":    _col(d, RED),
        "REVIEW":   _col(d, YEL),
        "NO_ALERT": _col(d, GRN),
    }.get(d, d)

def _make_dataset(n: int, ds_code: str, seed: int = 42) -> pd.DataFrame:
    rng = random.Random(seed)
    recs = []
    for i in range(n):
        name, nat, dob = rng.choice(_WATCHLIST)
        # 25% chance of token-reversed alias stored
        parts = name.split()
        alias = " ".join(reversed(parts)) if len(parts) > 1 and rng.random() < 0.25 else ""
        recs.append({
            "entity_id":      f"{ds_code}_{i:04d}",
            "dataset_type":   ds_code,
            "entity_type":    "INDIVIDUAL",
            "primary_name":   name,
            "aliases":        alias,
            "dob":            dob,
            "gender":         rng.choice(["M", "F"]),
            "nationality":    nat,
            "id_number":      f"XX{rng.randint(1_000_000, 9_999_999)}",
            "address_country": nat,
            "is_deceased":    False,
            "risk_weight":    round(rng.uniform(0.50, 1.00), 4),
            "risk_label":     int(rng.random() < 0.15),
        })
    return pd.DataFrame(recs)


class _MockRegistry:
    """Self-contained registry using synthetic data — no faker needed."""

    def __init__(self, n_per_dataset: int = 70):
        self._store: dict[tuple, pd.DataFrame] = {}
        for i, ds in enumerate(_DATASETS):
            self._store[(ds, "INDIVIDUAL")] = _make_dataset(n_per_dataset, ds, seed=i * 13 + 7)
        from data_layer.generators.ctry import generate
        self._ctry_df = generate()

    def get(self, dataset_code: str, entity_type: str | None = None) -> pd.DataFrame:
        c = dataset_code.upper()
        if entity_type:
            return self._store.get((c, entity_type.upper()), pd.DataFrame())
        parts = [df for (cc, _), df in self._store.items() if cc == c]
        return pd.concat(parts, ignore_index=True) if parts else pd.DataFrame()

    def list_datasets(self) -> list[tuple]:
        return [(c, e, len(df)) for (c, e), df in sorted(self._store.items())]

    def total_records(self) -> int:
        return sum(len(df) for df in self._store.values())

    def all_records_combined(self) -> pd.DataFrame:
        return pd.concat(list(self._store.values()), ignore_index=True)

class _MLEngine:
    """
    Drop-in replacement for ml_engine.model.AMLModel using sklearn GBT.
    Uses identical 22-feature vector from feature_engine.feature_builder.
    """

    def __init__(self):
        from sklearn.ensemble import GradientBoostingClassifier
        self._clf = GradientBoostingClassifier(
            n_estimators=80, max_depth=3, learning_rate=0.08,
            subsample=0.8, random_state=42,
        )
        self._trained = False

    def train(self, X: np.ndarray, y: np.ndarray) -> None:
        if y.sum() == 0:
            # no positives: can't train meaningfully — use dummy
            return
        self._clf.fit(X, y)
        self._trained = True
        pos = int(y.sum())
        neg = int(len(y) - pos)
        print(f"    [MLEngine] Trained on {len(y)} rows  "
              f"(positives={pos}, negatives={neg})")

    def predict_proba(self, x_row: np.ndarray) -> float:
        """Return probability of positive class for one feature row."""
        if not self._trained:
            return 0.5   # neutral fallback
        return float(self._clf.predict_proba(x_row.reshape(1, -1))[0, 1])

    def feature_importances(self) -> dict[str, float]:
        if not self._trained:
            return {}
        from feature_engine.feature_builder import FEATURE_NAMES
        return dict(zip(FEATURE_NAMES, self._clf.feature_importances_.round(4)))


def _build_training_data(registry: _MockRegistry) -> tuple[np.ndarray, np.ndarray]:
    """
    Build (X, y) training matrix from the registry.
    Positive class  = risk_label=1 records matched against themselves.
    Negative class  = random mismatched pairs.
    """
    from feature_engine.feature_builder import build_features, FEATURE_NAMES
    rng = random.Random(99)
    all_df = registry.all_records_combined()
    records = all_df.to_dict("records")

    X_rows, y_labels = [], []

    # Positives
    pos_records = [r for r in records if r.get("risk_label", 0) == 1]
    for rec in pos_records:
        row = pd.Series(rec)
        inp = {"name": rec["primary_name"], "dob": rec["dob"], "country": rec["nationality"]}
        mr  = {"composite_score": 1.0, "score_token_sort": 1.0, "score_token_set": 1.0,
               "score_partial": 0.95, "score_jaro_winkler": 0.98, "matched_on_alias": False}
        f   = build_features(match_result=mr, input_record=inp, candidate_row=row)
        X_rows.append([f[k] for k in FEATURE_NAMES])
        y_labels.append(1)

    # Negatives (random cross-pairs)
    n_neg = max(len(pos_records) * 4, 40)
    for _ in range(n_neg):
        ra, rb = rng.choice(records), rng.choice(records)
        if ra["entity_id"] == rb["entity_id"]:
            continue
        row = pd.Series(ra)
        score = rng.uniform(0.0, 0.50)
        inp = {"name": rb["primary_name"], "dob": rb["dob"], "country": rb["nationality"]}
        mr  = {"composite_score": score, "score_token_sort": score * 0.9,
               "score_token_set": score, "score_partial": score * 0.85,
               "score_jaro_winkler": score * 0.95, "matched_on_alias": False}
        f   = build_features(match_result=mr, input_record=inp, candidate_row=row)
        X_rows.append([f[k] for k in FEATURE_NAMES])
        y_labels.append(0)

    return np.array(X_rows, dtype=np.float32), np.array(y_labels, dtype=np.int32)

# ── 10a. Same rule results as left side (no need to re-query) ─────────────────
cr_int = cr_std  # same consolidation — blending happens below

# ── 10b. Feature Engine ───────────────────────────────────────────────────────
ml_score_val      = None
graph_risk_val    = None
blended_score_val = None
ml_features_dict  = None
top_cand_row      = None

if cr_int.top_hit is not None:
    th = cr_int.top_hit

    # Find the actual DataFrame row for the top hit entity
    for ds_code in _DATASETS:
        df_check = registry.get(ds_code, "INDIVIDUAL")
        if df_check.empty:
            continue
        mask = df_check["entity_id"] == th.entity_id
        if mask.any():
            top_cand_row = df_check[mask].iloc[0]
            break

    if top_cand_row is not None:
        # Build match_result dict from stored rule scores
        ars = th.all_rule_scores or {}
        composite = th.name_score
        mr_dict = {
            "composite_score":    composite,
            "score_token_sort":   ars.get("I020O", composite),
            "score_token_set":    ars.get("I020O", composite),
            "score_partial":      ars.get("I020O", composite) * 0.90,
            "score_jaro_winkler": ars.get("I020O", composite) * 0.95,
            "matched_on_alias":   th.matched_on_alias,
        }

        # ── 10c. 22-feature extraction ─────────────────────────────────────
        ml_features_dict = build_features(
            match_result=mr_dict,
            input_record=inp_dict,
            candidate_row=top_cand_row,
            rule_result=th,
        )

        # ── 10d. ML Engine inference ───────────────────────────────────────
        feat_vec  = np.array([[ml_features_dict[k] for k in FEATURE_NAMES]], dtype=np.float32)
        ml_score_val = ml_engine.predict_proba(feat_vec[0])

        # ── 10e. Graph Engine risk lookup ──────────────────────────────────
        eid = th.entity_id
        if G.has_node(eid):
            graph_risk_val = float(G.nodes[eid]["risk"])
        else:
            graph_risk_val = float(top_cand_row.get("risk_weight", 0.5))

        # ── 10f. Score Blending ────────────────────────────────────────────
        #   50% rule-based   +   30% ML   +   20% graph risk
        RULE_W, ML_W, GRAPH_W = 0.50, 0.30, 0.20
        blended_score_val = min(1.0, round(
            cr_int.consolidated_score * RULE_W
            + ml_score_val             * ML_W
            + graph_risk_val           * GRAPH_W,
            4,
        ))

# Final decision on blended score (or rule score if no top hit)
final_score_int = blended_score_val if blended_score_val is not None else cr_int.consolidated_score
dec_int         = apply_threshold(final_score_int, job_code)

# Audit + alert (integrated)

if cr_int.top_hit:
    log_decision(cr_int.top_hit.entity_id, inp.name,
                 cr_int.top_hit.matched_name, dec_int, "HIGH")
    a = create_alert(cr_int.top_hit.entity_id, inp.name,
                     cr_int.top_hit.matched_name, dec_int)
    if a:
        int_alert_ids.append(a.alert_id)

t_int_ms = (time.perf_counter() - t_int_start) * 1000

# ── 10g. KPI tracking ─────────────────────────────────────────────────────────
kpi_tracker.record(t_int_ms / 1000.0)
if dec_int.decision == "ALERT":
    kpi_tp += 1
elif dec_int.decision == "REVIEW":
    kpi_tp += 1
else:
    kpi_tn += 1
