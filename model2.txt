"""
ml_engine/model.py
──────────────────
AML name-match scoring model.

Changed from original:
  - Replaced xgboost with sklearn GradientBoostingClassifier (xgboost not installed)
  - Same public API: AMLModel.train(X, y) / .predict(X) / .evaluate(X, y) / .save() / .load()
  - Added build_training_data(registry) — builds X, y from the registry so the
    orchestrator can call model.train(*AMLModel.build_training_data(registry))
    without knowing about features or sample construction
  - Added cv_metrics property — populated after train(); used by cli/screen.py report
  - Added predict_one(features_dict) — single-record convenience used by orchestrator
  - Kept all original methods unchanged: predict_labels, evaluate, save, load
"""

from __future__ import annotations
import os
import json
import random
from typing import Any

import numpy as np
import pandas as pd
from sklearn.ensemble import GradientBoostingClassifier
from sklearn.model_selection import StratifiedKFold, cross_validate
from sklearn.metrics import (
    roc_auc_score, average_precision_score,
    classification_report, confusion_matrix,
)

from feature_engine.feature_builder import build_features, FEATURE_NAMES
from policy_engine.thresholds import apply_threshold


class AMLModel:
    """
    GradientBoosting classifier for AML name-match scoring.

    Workflow
    --------
    1.  model     = AMLModel()
    2.  X, y      = AMLModel.build_training_data(registry)
    3.  model.train(X, y)
    4.  score     = model.predict_one(features_dict)   # single record
    5.  scores    = model.predict(X)                   # batch
    6.  decisions = model.predict_labels(X)            # 'ALERT' / 'NO_ALERT'
    7.  metrics   = model.evaluate(X, y)               # held-out eval
    8.  model.save('model.joblib')
    9.  model     = AMLModel.load('model.joblib')
    """

    def __init__(
        self,
        n_estimators:  int   = 100,
        max_depth:     int   = 3,
        learning_rate: float = 0.05,
        subsample:     float = 0.8,
        random_state:  int   = 42,
    ) -> None:
        self.n_estimators  = n_estimators
        self.max_depth     = max_depth
        self.learning_rate = learning_rate
        self.subsample     = subsample
        self.random_state  = random_state
        self.feature_names = list(FEATURE_NAMES)
        self._model: GradientBoostingClassifier | None = None
        self._is_trained: bool = False
        self._cv_metrics: dict = {}       # populated after train()

    # ── Training data builder ────────────────────────────────────────────────

    @staticmethod
    def build_training_data(registry) -> tuple[np.ndarray, np.ndarray]:
        """
        Build feature matrix X and label vector y directly from the registry.

        Positive samples = self-match of every record with risk_label=1.
        Negative samples = cross-matched pairs (different entities, low score).

        Returns
        -------
        X : np.ndarray  shape (n_samples, n_features)  dtype float32
        y : np.ndarray  shape (n_samples,)              dtype int32
        """
        rng = random.Random(7)

        all_dfs = [
            df for (_, etype), df in registry._store.items()
            if etype == "INDIVIDUAL"
        ]
        if not all_dfs:
            return (
                np.zeros((0, len(FEATURE_NAMES)), dtype=np.float32),
                np.zeros(0, dtype=np.int32),
            )

        all_df  = pd.concat(all_dfs, ignore_index=True)
        records = all_df.to_dict("records")
        pos     = [r for r in records if r.get("risk_label", 0) == 1]
        neg     = [r for r in records if r.get("risk_label", 0) == 0]

        X_rows, y_labels = [], []

        # Positives: self-match with near-perfect feature scores
        for rec in pos:
            row = pd.Series(rec)
            inp = {
                "name":    rec["primary_name"],
                "dob":     str(rec.get("dob", "") or ""),
                "country": str(rec.get("nationality", "") or ""),
            }
            mr = {
                "composite_score":    1.0,
                "score_token_sort":   1.0,
                "score_token_set":    1.0,
                "score_partial":      0.95,
                "score_jaro_winkler": 0.98,
                "matched_on_alias":   False,
            }
            f = build_features(match_result=mr, input_record=inp, candidate_row=row)
            X_rows.append([f[k] for k in FEATURE_NAMES])
            y_labels.append(1)

        # Negatives: cross-matched non-match pairs with simulated low scores
        n_neg = max(len(pos) * 4, 60)
        for _ in range(n_neg):
            ra = rng.choice(neg)
            rb = rng.choice(pos + neg)
            if ra["entity_id"] == rb.get("entity_id"):
                continue
            score = rng.uniform(0.0, 0.45)
            inp   = {
                "name":    rb["primary_name"],
                "dob":     str(rb.get("dob", "") or ""),
                "country": str(rb.get("nationality", "") or ""),
            }
            mr = {
                "composite_score":    score,
                "score_token_sort":   score * 0.9,
                "score_token_set":    score,
                "score_partial":      score * 0.85,
                "score_jaro_winkler": score * 0.95,
                "matched_on_alias":   False,
            }
            f = build_features(
                match_result=mr, input_record=inp, candidate_row=pd.Series(ra)
            )
            X_rows.append([f[k] for k in FEATURE_NAMES])
            y_labels.append(0)

        return (
            np.array(X_rows,  dtype=np.float32),
            np.array(y_labels, dtype=np.int32),
        )

    # ── Train ────────────────────────────────────────────────────────────────

    def train(
        self,
        X: pd.DataFrame | np.ndarray,
        y: pd.Series   | np.ndarray,
        eval_fraction: float = 0.15,   # kept for API compatibility
    ) -> "AMLModel":
        """
        Train the model.

        Runs 5-fold CV first to produce honest out-of-sample metrics (stored in
        self.cv_metrics), then fits on the full data for deployment.
        """
        y_arr = np.asarray(y)
        n_pos = int(y_arr.sum())
        n_neg = len(y_arr) - n_pos

        if n_pos == 0:
            raise ValueError(
                "Training data has no positive examples (risk_label=1). "
                "Check dataset generation."
            )

        print(f"[AMLModel] {n_neg} negatives, {n_pos} positives")

        clf = GradientBoostingClassifier(
            n_estimators=self.n_estimators,
            max_depth=self.max_depth,
            learning_rate=self.learning_rate,
            subsample=self.subsample,
            random_state=self.random_state,
        )

        # 5-fold CV for honest metrics
        cv = StratifiedKFold(n_splits=5, shuffle=True, random_state=self.random_state)
        cv_res = cross_validate(
            clf, X, y_arr, cv=cv,
            scoring=["roc_auc", "average_precision", "f1", "precision", "recall"],
        )

        # Full-data fit for deployment
        clf.fit(X, y_arr)
        self._model      = clf
        self._is_trained = True

        y_prob = clf.predict_proba(X)[:, 1]
        y_pred = (y_prob >= 0.5).astype(int)
        cm     = confusion_matrix(y_arr, y_pred)

        self._cv_metrics = {
            "n_samples":    len(y_arr),
            "n_pos":        n_pos,
            "n_neg":        n_neg,
            "cv_roc_auc":   round(float(cv_res["test_roc_auc"].mean()),           4),
            "cv_roc_std":   round(float(cv_res["test_roc_auc"].std()),            4),
            "cv_pr_auc":    round(float(cv_res["test_average_precision"].mean()), 4),
            "cv_f1":        round(float(cv_res["test_f1"].mean()),                4),
            "cv_precision": round(float(cv_res["test_precision"].mean()),         4),
            "cv_recall":    round(float(cv_res["test_recall"].mean()),            4),
            "train_auc":    round(float(roc_auc_score(y_arr, y_prob)),           4),
            "cm":           cm.tolist(),
            "cls_report":   classification_report(
                                y_arr, y_pred,
                                target_names=["CLEAR", "ALERT"], digits=4),
            "feature_importances": {
                k: round(float(v), 4)
                for k, v in zip(FEATURE_NAMES, clf.feature_importances_)
            },
        }
        return self

    # ── Predict ──────────────────────────────────────────────────────────────

    def predict(self, X: pd.DataFrame | np.ndarray) -> np.ndarray:
        """Return probability scores for a batch. Shape (n_samples,)."""
        self._assert_trained()
        return self._model.predict_proba(X)[:, 1]   # type: ignore[union-attr]

    def predict_one(self, features: dict) -> float:
        """
        Score a single feature dict. Returns float [0-1].
        Used by ScreeningOrchestrator._screen_integrated().
        """
        self._assert_trained()
        vec = np.array(
            [[features.get(k, 0.0) for k in self.feature_names]],
            dtype=np.float32,
        )
        return float(self._model.predict_proba(vec)[0, 1])   # type: ignore[union-attr]

    def predict_labels(
        self,
        X: pd.DataFrame | np.ndarray,
        threshold: float | None = None,
    ) -> list[str]:
        """Return 'ALERT' / 'NO_ALERT' decision for each row in X."""
        scores = self.predict(X)
        return [apply_threshold(float(s), threshold=threshold).decision for s in scores]

    # ── Evaluate ─────────────────────────────────────────────────────────────

    def evaluate(
        self,
        X: pd.DataFrame | np.ndarray,
        y: pd.Series   | np.ndarray,
    ) -> dict[str, Any]:
        """
        Compute metrics on a held-out set.
        Prints summary to stdout (matches original xgboost version behaviour).
        """
        self._assert_trained()
        y_arr    = np.asarray(y)
        y_scores = self.predict(X)
        y_pred   = (y_scores >= 0.5).astype(int)

        metrics = {
            "roc_auc":               round(float(roc_auc_score(y_arr, y_scores)), 4),
            "pr_auc":                round(float(average_precision_score(y_arr, y_scores)), 4),
            "classification_report": classification_report(
                                        y_arr, y_pred, target_names=["CLEAR", "ALERT"]),
            "confusion_matrix":      confusion_matrix(y_arr, y_pred).tolist(),
        }
        print(f"ROC-AUC : {metrics['roc_auc']}")
        print(f"PR-AUC  : {metrics['pr_auc']}")
        print(metrics["classification_report"])
        return metrics

    @property
    def cv_metrics(self) -> dict:
        """5-fold CV metrics. Populated after train(). Read by cli/screen.py."""
        return self._cv_metrics

    # ── Persistence ──────────────────────────────────────────────────────────

    def save(self, path: str) -> None:
        """Persist model to disk using joblib."""
        import joblib
        self._assert_trained()
        joblib.dump(self._model, path)
        meta = {
            "feature_names": self.feature_names,
            "cv_metrics":    self._cv_metrics,
        }
        with open(path + ".meta.json", "w") as f:
            json.dump(meta, f, indent=2)
        print(f"[AMLModel] Saved to {path}")

    @classmethod
    def load(cls, path: str) -> "AMLModel":
        """Load a previously saved model."""
        import joblib
        instance            = cls()
        instance._model     = joblib.load(path)
        meta_path = path + ".meta.json"
        if os.path.exists(meta_path):
            with open(meta_path) as f:
                meta = json.load(f)
            instance.feature_names = meta.get("feature_names", list(FEATURE_NAMES))
            instance._cv_metrics   = meta.get("cv_metrics", {})
        instance._is_trained = True
        return instance

    # ── Internal ─────────────────────────────────────────────────────────────

    def _assert_trained(self) -> None:
        if not self._is_trained or self._model is None:
            raise RuntimeError("Model not trained. Call .train(X, y) first.")
